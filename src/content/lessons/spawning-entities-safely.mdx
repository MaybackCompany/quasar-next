## Build it

<StepGroup>
  <Step num={1} title="Make the resource folder" outcome="The server has one folder for this lesson.">
    Inside your server's `resources` folder, create this folder:

    ```text
    resources/qu_spawning_entities_safely
    ```
  </Step>

  <Step num={2} title="Create the files" outcome="Every file named in the manifest exists.">
    Create this exact file layout:

    ```text
resources/qu_spawning_entities_safely/
  fxmanifest.lua
  client.lua
    ```
  </Step>

  <Step num={3} title="Write fxmanifest.lua" outcome="FiveM knows which files to load.">
    Open `fxmanifest.lua` and paste this:

    ```lua
fx_version 'cerulean'
game 'gta5'

client_script 'client.lua'
    ```
  </Step>

  <Step num={4} title="Write the lesson code" outcome="The topic is now represented by runnable code.">
    Open `client.lua` and paste this:

    ```lua
local objects = {}

RegisterCommand('spawnbox', function()
    local model = joaat('prop_boxpile_07d')
    RequestModel(model)
    while not HasModelLoaded(model) do Wait(0) end

    local coords = GetEntityCoords(PlayerPedId())
    local object = CreateObject(model, coords.x + 2.0, coords.y, coords.z, true, true, false)
    objects[object] = true
    SetModelAsNoLongerNeeded(model)
    print('[qu_spawning_entities_safely] spawned object ' .. object)

    SetTimeout(30000, function()
        if DoesEntityExist(object) then
            DeleteEntity(object)
            objects[object] = nil
            print('[qu_spawning_entities_safely] timeout deleted object')
        end
    end)
end, false)

RegisterCommand('clearboxes', function()
    for object in pairs(objects) do
        if DoesEntityExist(object) then DeleteEntity(object) end
        objects[object] = nil
    end
    print('[qu_spawning_entities_safely] registry cleared')
end, false)

AddEventHandler('onResourceStop', function(name)
    if name == GetCurrentResourceName() then
        ExecuteCommand('clearboxes')
    end
end)
    ```
  </Step>

  <Step num={5} title="Start and test it" outcome="The expected proof appears in the correct console.">
    Open `server.cfg` and add this line:

    ```text
    ensure qu_spawning_entities_safely
    ```

    Save, then run:

    ```text
    restart qu_spawning_entities_safely
    ```

    Run this test:

    ```text
    /spawnbox then /clearboxes
    ```

    <ExpectedOutput>
      F8: [qu_spawning_entities_safely] spawned object
F8: [qu_spawning_entities_safely] registry cleared
    </ExpectedOutput>
  </Step>
</StepGroup>
